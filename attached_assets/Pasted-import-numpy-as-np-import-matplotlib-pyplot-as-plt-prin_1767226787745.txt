import numpy as np
import matplotlib.pyplot as plt

print("=== RAI GRUT FULL VALIDATION SUITE ===")

# --- Constants & GRUT arrays (ensure these are defined from your engine output) ---
# Example placeholders (replace with actual engine values)
sigma8_0 = 0.811
Omega_b = 0.0486
Omega_geom = 0.25
alpha = -1/12
tau0 = 41.9e6 * 365 * 24 * 3600  # seconds

# Arrays: z_array descending (high-z → low-z), t_array ascending (cosmic time)
# D, f, G_eff, Omega_kernel should be arrays of same length as z_array
# For demonstration, we use placeholder arrays; replace with engine outputs
z_array = np.linspace(10, 0, 1000)  # descending
t_array = np.linspace(0, 13.8e9*3.154e7, 1000)  # seconds, cosmic time
D = np.linspace(0.01, 1.0, 1000)  # placeholder growth factor
f = np.linspace(0.05, 0.5, 1000)  # placeholder growth rate
G_eff = np.linspace(1.0, 4/3, 1000)
Omega_kernel = np.linspace(0.001, 0.06, 1000)

# --- Observational Data ---
z_obs_array = np.array([0.15, 0.38, 0.51, 0.70, 1.48, 2.5])
fsigma8_obs_array = np.array([0.49, 0.44, 0.45, 0.47, 0.46, 0.46])
sigma_obs_array = np.array([0.05, 0.04, 0.04, 0.04, 0.04, 0.04])

# --- High-z test ---
z_early_array = np.array([5.0, 6.0, 7.0, 8.0, 9.0, 10.0])

# --- Burst scenario ---
z_burst = 1.0
burst_factor = 2.0

# --- Helper function ---
def kernel_fraction(z_val, Omega_kernel_val):
    Omega_eff_val = Omega_b * (1 + z_val)**3 / (Omega_b * (1 + z_val)**3 + Omega_geom)
    return Omega_kernel_val / (Omega_eff_val + Omega_kernel_val)

# --- Initialize result lists ---
D_vals_obs, f_vals_obs, G_eff_vals_obs, Omega_kernel_frac_obs, fsigma8_vals_obs, chi2_terms = [], [], [], [], [], []
D_vals_early, f_vals_early, G_eff_vals_early, Omega_kernel_frac_early, fsigma8_vals_early, Phi_vals_early = [], [], [], [], [], []

# --- Observational fσ8 test ---
print("\n--- Observational fσ8 Test ---")
chi2_total = 0
for z_test, fsigma8_obs, sigma_obs in zip(z_obs_array, fsigma8_obs_array, sigma_obs_array):
    D_z = np.interp(z_test, z_array[::-1], D[::-1])
    f_z = np.interp(z_test, z_array[::-1], f[::-1])
    G_eff_z = np.interp(z_test, z_array[::-1], G_eff[::-1])
    Omega_kernel_z = np.interp(z_test, z_array[::-1], Omega_kernel[::-1])
    
    # Burst adjustment
    if abs(z_test - z_burst) < 0.01:
        Omega_kernel_z *= burst_factor

    fsigma8_z = f_z * sigma8_0 * D_z
    k_frac = kernel_fraction(z_test, Omega_kernel_z)
    chi2_term = ((fsigma8_z - fsigma8_obs) / sigma_obs) ** 2
    chi2_total += chi2_term

    D_vals_obs.append(D_z)
    f_vals_obs.append(f_z)
    G_eff_vals_obs.append(G_eff_z)
    Omega_kernel_frac_obs.append(k_frac)
    fsigma8_vals_obs.append(fsigma8_z)
    chi2_terms.append(chi2_term)

    print(f"z={z_test:.2f} fσ8={fsigma8_z:.4f}, Ω_kernel_frac={k_frac:.4f}, χ²={chi2_term:.2f}")

print(f"\nTotal χ² (observations) = {chi2_total:.2f} (Physics-compliant floor ~15–16)")

# --- High-z Early Universe Test ---
print("\n--- High-z Early Universe Test ---")
chi2_high_total = 0
for z_test in z_early_array:
    D_z = np.interp(z_test, z_array[::-1], D[::-1])
    f_z = np.interp(z_test, z_array[::-1], f[::-1])
    G_eff_z = np.interp(z_test, z_array[::-1], G_eff[::-1])
    Omega_kernel_z = np.interp(z_test, z_array[::-1], Omega_kernel[::-1])
    
    fsigma8_z = f_z * sigma8_0 * D_z
    k_frac = kernel_fraction(z_test, Omega_kernel_z)

    # Integrated potential Φ(z)
    idx = np.searchsorted(z_array[::-1], z_test)
    delta_t = t_array[idx] - t_array[:idx+1]
    kernel_vals = 12 * (abs(alpha)/tau0) * np.exp(-delta_t/tau0) * (delta_t >= 0)
    Phi_z = np.sum(kernel_vals * Omega_b * (1 + z_array[:idx+1])**3 * np.diff(np.append(0, t_array[:idx+1])))

    chi2_high_total += ((fsigma8_z - 0.42)/0.04) ** 2

    D_vals_early.append(D_z)
    f_vals_early.append(f_z)
    G_eff_vals_early.append(G_eff_z)
    Omega_kernel_frac_early.append(k_frac)
    fsigma8_vals_early.append(fsigma8_z)
    Phi_vals_early.append(Phi_z)

    print(f"z={z_test:.2f} fσ8={fsigma8_z:.4f}, Ω_kernel_frac={k_frac:.4f}, Φ(z)={Phi_z:.4e}")

print(f"\nTotal χ² (high-z test) = {chi2_high_total:.2f}")

# --- Low-z S8 check ---
D_z_low = np.interp(0.3, z_array[::-1], D[::-1])
f_z_low = np.interp(0.3, z_array[::-1], f[::-1])
S8_pred = sigma8_0 * D_z_low * (Omega_b / 0.3) ** 0.5
print(f"\nPredicted S8 at z~0.3 = {S8_pred:.3f}")

# --- Kernel fraction at z=10 ---
kernel_frac_highz = kernel_fraction(10.0, Omega_kernel[-1])
print(f"Kernel fraction at z=10 = {kernel_frac_highz:.4f} (should be minimal)")

# --- Visualizations ---
plt.figure(figsize=(12, 5))

# fσ8 observational plot
plt.subplot(1,3,1)
plt.scatter(z_obs_array, fsigma8_obs_array, color='red', label='Observed fσ8')
plt.plot(z_obs_array, fsigma8_vals_obs, label='GRUT Predicted fσ8')
plt.xlabel('Redshift z')
plt.ylabel('fσ8')
plt.title('fσ8 Comparison')
plt.legend()
plt.grid(True)

# G_eff plot

plt.xlabel('Redshift z')

plt.title('Effective Gravitational Coupling')
plt.grid(True)

# Φ(z) high-z potential plot
plt.subplot(1,3,3)
plt.plot(z_early_array, Phi_vals_early, label='Φ(z)', color='blue')
plt.xlabel('Redshift z')
plt.ylabel('Φ(z)')
plt.title('High-z Potential')
