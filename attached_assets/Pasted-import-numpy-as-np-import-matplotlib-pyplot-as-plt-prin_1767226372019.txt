import numpy as np
import matplotlib.pyplot as plt

print("=== FINAL RAI GRUT FULL STRESS TEST ===")

# -------------------------------
# Constants from GRUT engine
# -------------------------------
sigma8_0 = 0.811      # Present-day normalization
Omega_b = 0.0486      # Baryon density parameter
Omega_geom = 0.7      # Geometric stiffness (residual)
alpha = -1/12         # Vacuum tension coefficient
tau0 = 41.9e6 * 365*24*3600  # Relaxation time in seconds (~41.9 Myr)

# -------------------------------
# GRUT arrays (from engine)
# -------------------------------
# Example placeholder arrays — replace with engine outputs
# z_array descending (early → late)
z_array = np.linspace(10, 0, 1000)
D = np.linspace(0.01, 1, 1000)          # Growth factor
f = np.linspace(0.04, 0.5, 1000)        # Growth rate
G_eff = np.linspace(1.0, 4/3, 1000)     # Effective gravitational coupling
Omega_kernel = np.linspace(0.0, 0.02, 1000)  # Kernel fraction
t_array = np.linspace(0, 13.8e9*365*24*3600, 1000)  # Cosmic time in seconds

# -------------------------------
# Observational data
# -------------------------------
z_obs_array = np.array([0.15, 0.38, 0.51, 0.70, 1.48, 2.5])
fsigma8_obs_array = np.array([0.49, 0.44, 0.45, 0.47, 0.46, 0.46])
sigma_obs_array = np.array([0.05, 0.04, 0.04, 0.04, 0.04, 0.04])

z_early_array = np.array([5.0, 6.0, 7.0, 8.0, 9.0, 10.0])
z_burst = 1.0
burst_factor = 2.0

# -------------------------------
# Safe interpolation on descending z
# -------------------------------
def interp_z_safe(z_val, arr):
    return np.interp(z_val, z_array[::-1], arr[::-1])

# -------------------------------
# Kernel fraction function
# -------------------------------
def kernel_fraction(z_val, Omega_kernel_val):
    Omega_eff_val = Omega_b * (1 + z_val)**3 / (Omega_b*(1+z_val)**3 + Omega_geom)
    return Omega_kernel_val / (Omega_eff_val + Omega_kernel_val)

# -------------------------------
# Storage arrays
# -------------------------------
D_vals_obs, f_vals_obs, G_eff_vals_obs, Omega_kernel_frac_obs, fsigma8_vals_obs, chi2_terms = [], [], [], [], [], []
D_vals_early, f_vals_early, G_eff_vals_early, Omega_kernel_frac_early, fsigma8_vals_early, Phi_vals_early = [], [], [], [], [], []

# -------------------------------
# Step 1: Observational fσ8 Test
# -------------------------------
print("\n--- Observational fσ8 Test ---")
chi2_total = 0
for z_test, fsigma8_obs, sigma_obs in zip(z_obs_array, fsigma8_obs_array, sigma_obs_array):
    D_z = interp_z_safe(z_test, D)
    f_z = interp_z_safe(z_test, f)
    G_eff_z = interp_z_safe(z_test, G_eff)
    Omega_kernel_z = interp_z_safe(z_test, Omega_kernel)
    
    if abs(z_test - z_burst) < 0.01:
        Omega_kernel_z *= burst_factor

    fsigma8_z = f_z * sigma8_0 * D_z
    k_frac = kernel_fraction(z_test, Omega_kernel_z)
    chi2_term = ((fsigma8_z - fsigma8_obs)/sigma_obs)**2
    chi2_total += chi2_term

    D_vals_obs.append(D_z)
    f_vals_obs.append(f_z)
    G_eff_vals_obs.append(G_eff_z)
    Omega_kernel_frac_obs.append(k_frac)
    fsigma8_vals_obs.append(fsigma8_z)
    chi2_terms.append(chi2_term)

    print(f"z={z_test:.2f} fσ8={fsigma8_z:.4f}, Ω_kernel_frac={k_frac:.4f}, χ²={chi2_term:.2f}")

print(f"\nTotal χ² (observations) = {chi2_total:.2f} (Physics-compliant floor ~15–16)")

# -------------------------------
# Step 2: High-z Early Universe Test
# -------------------------------
print("\n--- High-z Early Universe Test ---")
chi2_high_total = 0
for z_test in z_early_array:
    D_z = interp_z_safe(z_test, D)
    f_z = interp_z_safe(z_test, f)
    G_eff_z = interp_z_safe(z_test, G_eff)
    Omega_kernel_z = interp_z_safe(z_test, Omega_kernel)

    fsigma8_z = f_z * sigma8_0 * D_z
    k_frac = kernel_fraction(z_test, Omega_kernel_z)

    idx = np.searchsorted(z_array[::-1], z_test)
    delta_t = t_array[idx] - t_array[:idx+1]
    kernel_vals = 12 * (abs(alpha)/tau0) * np.exp(-delta_t / tau0) * (delta_t>=0)
    Phi_z = np.sum(kernel_vals * Omega_b * (1+z_array[:idx+1])**3 * np.diff(np.append(0, t_array[:idx+1])))
    
    chi2_high_total += ((fsigma8_z - 0.42)/0.04)**2

    D_vals_early.append(D_z)
    f_vals_early.append(f_z)
    G_eff_vals_early.append(G_eff_z)
    Omega_kernel_frac_early.append(k_frac)
    fsigma8_vals_early.append(fsigma8_z)
    Phi_vals_early.append(Phi_z)

    print(f"z={z_test:.2f} fσ8={fsigma8_z:.4f}, Ω_kernel_frac={k_frac:.4f}, Φ(z)={Phi_z:.4e}")

print(f"\nTotal χ² (high-z test) = {chi2_high_total:.2f}")

# -------------------------------
# Step 3: Low-z S8 check
