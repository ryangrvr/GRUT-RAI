VI. MINIMAL VALIDATOR SKELETON (REFERENCE)
class GRUTConstitutionViolation(Exception):
    pass


class GRUTValidator:

    def run_all(self, engine):
        self.check_forbidden_shortcuts(engine)
        self.test_memory_priority(engine)
        self.test_catch_up(engine)
        self.test_high_z_guardrail(engine)
        self.test_isw_consistency(engine)

    def check_forbidden_shortcuts(self, engine):
        if engine.uses_local_growth_law():
            raise GRUTConstitutionViolation("Local growth law detected")

        if engine.uses_instantaneous_Geff():
            raise GRUTConstitutionViolation("Instantaneous G_eff detected")

    def test_memory_priority(self, engine):
        Phi_A = engine.simulate(universe="A", z=0.3)
        Phi_B = engine.simulate(universe="B", z=0.3)
        if Phi_A <= Phi_B:
            raise GRUTConstitutionViolation("Memory Priority violated")

    def test_catch_up(self, engine):
        if engine.detects_instant_catchup():
            raise GRUTConstitutionViolation("Instant catch-up forbidden")

    def test_high_z_guardrail(self, engine):
        if engine.G_eff(z=3.0) > engine.G:
            raise GRUTConstitutionViolation("High-z GR guardrail violated")

    def test_isw_consistency(self, engine):
        if not engine.satisfies_isw_inequality():
            raise GRUTConstitutionViolation("ISW overproduction detected")


This is not optional.
