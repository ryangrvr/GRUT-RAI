import streamlit as st
import plotly.graph_objects as go

def render_comparative_overlay(tau_0, velocity):
    st.markdown("#### 2. Observational Overlay: GRUT vs. Hubble/Chandra")
    
    # Validate inputs
    if tau_0 <= 0 or velocity < 0:
        st.error("Please provide valid positive values for Tau_0 and Velocity.")
        return

    # Calculate the physical drift based on the slider's Tau_0
    # Drift = v * tau_0 (converting tau_0 to years and then to Mly)
    drift_mly = (velocity * (tau_0 * 1e6 * 3.154e7)) / 9.461e15  # Convert to million light-years
    
    # Create the Overlay Plot
    fig = go.Figure()

    # Add the "Baryonic" Image (URL to the Bullet Cluster image)
    fig.add_layout_image(
        dict(
            source="https://upload.wikimedia.org/wikipedia/commons/b/b5/Bullet_cluster.jpg",
            xref="x", yref="y", x=-20, y=20, sizex=80, sizey=40,
            sizing="stretch", opacity=0.7, layer="below"
        )
    )

    # 1. Plot the Gas (Red - Where the mass IS)
    fig.add_trace(go.Scatter(
        x=[0], y=[0],
        mode='markers',
        marker=dict(size=15, color='red', symbol='x'),
        name="Baryonic Gas (X-Ray)"
    ))

    # 2. Plot the GRUT Prediction (Blue - Where the gravity Lags)
    fig.add_trace(go.Scatter(
        x=[drift_mly], y=[0],
        mode='markers',
        marker=dict(size=25, color='cyan', line=dict(width=2, color='white')),
        name="GRUT Lensing Prediction"
    ))

    # Updating the layout
    fig.update_layout(
        xaxis=dict(showgrid=False, zeroline=False, range=[-20, 60], title="Distance (Mly)"),
        yaxis=dict(showgrid=False, zeroline=False, range=[-20, 20], title="Y Axis"),
        height=400,
        margin=dict(l=0, r=0, b=0, t=0),
        legend=dict(yanchor="top", y=0.99, xanchor="left", x=0.01)
    )

    # Render the plot in Streamlit
    st.plotly_chart(fig, use_container_width=True)
    st.caption(f"At {tau_0} Myr, the 'Memory Ghost' shifts {drift_mly:.2f} Mly from the baryonic center.")
